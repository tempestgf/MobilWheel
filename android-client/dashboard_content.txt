    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_dashboard)
        window.addFlags(WindowManager.LayoutParams.FLAG_KEEP_SCREEN_ON)

        initializeUIElements()
        setupSensors()
        setConnectionState(false, null)
        showConnectionOptions()
        startStatusUpdates()
    }

    private fun startStatusUpdates() {
        statusUpdateJob = CoroutineScope(Dispatchers.Main).launch {
            val timeFormat = SimpleDateFormat("HH:mm", Locale.getDefault())
            while (isActive) {
                val now = Date()
                timeStatusText.text = timeFormat.format(now)
                
                val batteryStatus: Intent? = IntentFilter(Intent.ACTION_BATTERY_CHANGED).let { ifilter ->
                    applicationContext.registerReceiver(null, ifilter)
                }
                val level: Int = batteryStatus?.getIntExtra(BatteryManager.EXTRA_LEVEL, -1) ?: -1
                val scale: Int = batteryStatus?.getIntExtra(BatteryManager.EXTRA_SCALE, -1) ?: -1
                val batteryPct = if (level != -1 && scale != -1) (level * 100 / scale.toFloat()).toInt() else -1
                
                if (batteryPct != -1) {
                    batteryStatusText.text = "BAT "+batteryPct+"%"
                    if (batteryPct <= 15) {
                        batteryStatusText.setTextColor(0xFFFF6B6B.toInt())
                    } else {
                        batteryStatusText.setTextColor(0x55FFFFFF.toInt())
                    }
                }
                
                if (isConnected()) {
                    wifiStatusText.text = "PING <5ms"
                    wifiStatusText.setTextColor(0xFF4ADE80.toInt())
                } else {
                    wifiStatusText.text = "PING --"
                    wifiStatusText.setTextColor(0x55FFFFFF.toInt())
                }
                
                delay(10000)
            }
        }
    }

    private fun setConnectionState(connected: Boolean, ip: String?) {
        runOnUiThread {
            if (connected && ip != null) {
                connectionLabel.text = getString(R.string.title_connected)
                connectionLabel.setTextColor(0xFF4ADE80.toInt())
                connectionDot.setBackgroundResource(R.drawable.cockpit_dot_green)
                ipAddressDisplay.text = ip
            } else {
                connectionLabel.text = getString(R.string.title_offline)
                connectionLabel.setTextColor(0xFFFF6B6B.toInt())
                connectionDot.background = null
                connectionDot.setBackgroundColor(0x66FF6B6B.toInt())
                ipAddressDisplay.text = "-.-.-.-"
                
                speedValueView.text = "0"
                rpmIndicator.text = "0 RPM"
                gearIndicatorView.text = "N"
                rpmDots.forEach { it.alpha = 0.15f }
                ffbBars.forEach { it.alpha = 0.2f }
                updatePedal(accelerateIndicator, 0f)
                updatePedal(brakeIndicator, 0f)
            }
        }
    }

    private fun updatePedal(indicator: View, percentage: Float) {
        val params = indicator.layoutParams
        params.height = (mainView.height * percentage).toInt()
        indicator.layoutParams = params
    }

    private fun startTcpReceiver() {
        CoroutineScope(Dispatchers.IO).launch {
            try {
                val reader = socket?.getInputStream()?.bufferedReader() ?: return@launch
                while (isConnected()) {
                    try {
                        val line = reader.readLine() ?: break
                        if (line.startsWith("T:")) {
                            parseLegacyTelemetry(line)
                        } else if (line.startsWith("J:")) {
                            parseJsonTelemetry(line.substring(2))
                        } else if (line.startsWith("{")) {
                            parseJsonTelemetry(line)
                        }
                    } catch (e: Exception) {
                        break
                    }
                }
            } catch (e: Exception) {
                Log.e("TcpReceiver", "Error", e)
            } finally {
                setConnectionState(false, null)
            }
        }
    }

    private fun parseLegacyTelemetry(line: String) {
        val parts = line.split(":")
        if (parts.size != 4) return
        val speed = parts[1].toIntOrNull() ?: return
        val gearRaw = parts[2].toIntOrNull() ?: return
        val rpm = parts[3].toIntOrNull() ?: return

        val json = JSONObject()
        json.put("speed_kmh", speed)
        json.put("gear", gearRaw - 1)
        json.put("rpm", rpm)
        updateDashboard(json)
    }

    private fun parseJsonTelemetry(jsonStr: String) {
        try {
            val json = JSONObject(jsonStr)
            updateDashboard(json)
        } catch (e: Exception) {
            Log.e("JSON", "Error parsing log", e)
        }
    }

    private fun updateDashboard(json: JSONObject) {
        val speed = json.optDouble("speed_kmh", 0.0).toInt()
        val rpm = json.optInt("rpm", 0)
        val gear = json.optInt("gear", 1) // 0=N
        val maxRpm = json.optInt("max_rpm", 9000)
        
        val ffb = json.optDouble("ffb", 0.0)
        val gas = json.optDouble("gas", 0.0).toFloat()
        val brake = json.optDouble("brake", 0.0).toFloat()

        // Just in case it comes as throttle
        val actualGas = if (json.has("gas")) gas else json.optDouble("throttle", 0.0).toFloat()

        val rpmFraction = if (maxRpm > 0) rpm.toFloat() / maxRpm else 0f

        val thresholds = floatArrayOf(
            0.60f, 0.63f, 0.66f, 0.69f, 0.72f,
            0.75f, 0.77f, 0.79f, 0.81f, 0.83f,
            0.85f, 0.88f, 0.92f, 0.96f, 0.98f
        )

        runOnUiThread {
            speedValueView.text = speed.toString()

            // Advanced Telemetry (SmoothOperator integration)
            fun formatMs(ms: Int): String {
                if (ms <= 0) return "--:--.---"
                val min = ms / 60000
                val sec = (ms % 60000) / 1000
                val mil = ms % 1000
                return String.format("%02d:%02d.%03d", min, sec, mil)
            }
            
            // Fuel
            val fuel = json.optDouble("fuel", 0.0)
            tvFuel.text = String.format("FUEL %.1f L", fuel)
            
            // Laps (Use either camelCase or snake_case depending on backend version)
            val currentLap = json.optInt("currentLapMs", json.optInt("current_lap_ms", 0))
            val bestLap = json.optInt("bestLapMs", json.optInt("best_lap_ms", 0))
            tvLapTime.text = "LAP " + formatMs(currentLap)
            tvBestLap.text = "BEST " + formatMs(bestLap)

            // Tyres
            val tyreTemps = json.optJSONArray("tyre_core_temperature") ?: json.optJSONArray("tyreTemp")
            if (tyreTemps != null && tyreTemps.length() >= 4) {
                tvTyreFL.text = tyreTemps.optInt(0).toString()
                tvTyreFR.text = tyreTemps.optInt(1).toString()
                tvTyreRL.text = tyreTemps.optInt(2).toString()
                tvTyreRR.text = tyreTemps.optInt(3).toString()
                
                // Color code tyres based on optimal temps (AC GT3 ~ 80C)
                fun setTyreColor(tv: TextView, temp: Int) {
                    if (temp < 60) tv.setTextColor(0xFF7AA2F7.toInt()) // Cold (Blue)
                    else if (temp > 105) tv.setTextColor(0xFFFF6B6B.toInt()) // Hot (Red)
                    else tv.setTextColor(0xFF9ECE6A.toInt()) // Optimum (Green)
                }
                setTyreColor(tvTyreFL, tyreTemps.optInt(0))
                setTyreColor(tvTyreFR, tyreTemps.optInt(1))
                setTyreColor(tvTyreRL, tyreTemps.optInt(2))
                setTyreColor(tvTyreRR, tyreTemps.optInt(3))
            }
            
            // Position
            val pos = json.optInt("position", 0)
            val numCars = json.optInt("numCars", json.optInt("num_cars", 0))
            
            if (pos > 0) {
                tvPosition.text = "POS $pos / ${if (numCars>0) numCars else "--"}"
            }
            
            // Map Position Update
            val mapCoords = json.optJSONArray("tyre_contact_point")
            if (mapCoords != null && mapCoords.length() >= 3) {
                // FL tyre is [0, 1, 2] -> X, Y, Z
                val carX = mapCoords.getDouble(0).toFloat()
                val carZ = mapCoords.getDouble(2).toFloat()
                trackMapView.updatePosition(carX, carZ)
            }

            rpmIndicator.text = " RPM"
            gearIndicatorView.text = when {
                gear <= -1 -> "R"
                gear == 0 -> "N"
                else -> gear.toString()
            }

            rpmDots.forEachIndexed { i, dot ->
                if (i < thresholds.size) {
                    val lit = rpmFraction >= thresholds[i]
                    dot.alpha = if (lit) 1.0f else 0.15f
                }
            }

            // Update FFB representation (number of bars lit up)
            val litFfbBars = (Math.abs(ffb) * ffbBars.size).toInt().coerceIn(0, ffbBars.size)
            ffbBars.forEachIndexed { index, view ->
                view.alpha = if (index < litFfbBars) 1.0f else 0.2f
            }

            // Animate pedals
            updatePedal(accelerateIndicator, actualGas.coerceIn(0f, 1f))
            updatePedal(brakeIndicator, brake.coerceIn(0f, 1f))
            
            findViewById<View>(R.id.accelerateTopIndicator)?.visibility = if (actualGas >= 0.99f) View.VISIBLE else View.GONE
            findViewById<View>(R.id.brakeTopIndicator)?.visibility = if (brake >= 0.99f) View.VISIBLE else View.GONE
        }
    }

    @Synchronized
    private fun establishConnection() {
        if (isConnected()) return
        CoroutineScope(Dispatchers.IO).launch {
            try {
                closeConnection()
                socket = Socket(serverAddress, serverPort).apply {
                    tcpNoDelay = true
                    keepAlive = true
                    soTimeout = 5000
                }
                outputStream = DataOutputStream(socket?.getOutputStream())
                setConnectionState(true, serverAddress)
                startTcpReceiver()
            } catch (e: IOException) {
                setConnectionState(false, null)
                handleConnectionError(e)
            }
        }
    }

    private fun handleConnectionError(e: IOException) {
        runOnUiThread {
            Toast.makeText(this@DashboardActivity, "Connection Failed", Toast.LENGTH_LONG).show()
            showConnectionOptions()
        }
    }

    private fun closeConnection() {
        try {
            outputStream?.flush()
            socket?.close()
        } catch (e: Exception) {} finally {
            socket = null
            outputStream = null
        }
    }

    private fun isConnected() = socket?.isConnected == true && socket?.isClosed == false

    private fun discoverServerViaUDP() {
        CoroutineScope(Dispatchers.IO).launch {
            try {
                val udp = DatagramSocket()
                udp.broadcast = true
                val msg = "DISCOVER_SERVER".toByteArray()
                udp.send(DatagramPacket(msg, msg.size, InetAddress.getByName("255.255.255.255"), serverPort))
                val resp = DatagramPacket(ByteArray(1024), 1024)
                udp.soTimeout = 5000
                try {
                    udp.receive(resp)
                    serverAddress = String(resp.data, 0, resp.length).split(":")[0]
                    udp.close()
                    establishConnection()
                } catch (e: Exception) {
                    runOnUiThread { Toast.makeText(this@DashboardActivity, "PC Server not found.", Toast.LENGTH_SHORT).show(); showConnectionOptions() }
                }
            } catch (e: Exception) {
                runOnUiThread { Toast.makeText(this@DashboardActivity, "Network error.", Toast.LENGTH_SHORT).show(); showConnectionOptions() }
            }
        }
    }

    private fun discoverServerViaUsbTethering() {
        CoroutineScope(Dispatchers.IO).launch {
            try {
                val usbIp = getUsbTetheringIp()
                if (usbIp != null) {
                    val broadcast = ".255"
                    val udp = DatagramSocket()
                    udp.broadcast = true
                    val msg = "DISCOVER_SERVER".toByteArray()
                    udp.send(DatagramPacket(msg, msg.size, InetAddress.getByName(broadcast), serverPort))
                    val resp = DatagramPacket(ByteArray(1024), 1024)
                    udp.soTimeout = 3000
                    try {
                        udp.receive(resp)
                        serverAddress = String(resp.data, 0, resp.length).split(":")[0]
                        udp.close()
                        establishConnection()
                    } catch (e: Exception) {
                        runOnUiThread { Toast.makeText(this@DashboardActivity, "PC Server not found via USB.", Toast.LENGTH_SHORT).show(); showConnectionOptions() }
                    }
                } else {
                    withContext(Dispatchers.Main) { Toast.makeText(this@DashboardActivity, "USB Tethering unavailable.", Toast.LENGTH_SHORT).show() }
                }
            } catch (e: Exception) {
                runOnUiThread { Toast.makeText(this@DashboardActivity, "USB Network error.", Toast.LENGTH_SHORT).show(); showConnectionOptions() }
            }
        }
    }

    private fun getUsbTetheringIp(): String? = try {
        NetworkInterface.getNetworkInterfaces().toList()
            .firstOrNull { it.name.contains("rndis") || it.name.contains("usb") }
            ?.inetAddresses?.toList()
            ?.firstOrNull { !it.isLoopbackAddress && it is Inet4Address }
            ?.hostAddress
    } catch (e: Exception) { null }

    private fun showManualIpInputDialog() {
        val input = EditText(this).apply {
            inputType = InputType.TYPE_CLASS_PHONE
            setText(serverAddress)
        }
        MaterialAlertDialogBuilder(this, R.style.GeneonDialogTheme)
            .setTitle("Enter PC IP Address")
            .setView(input)
            .setPositiveButton("Connect") { _, _ ->
                serverAddress = input.text.toString()
                establishConnection()
            }
            .setNegativeButton("Cancel", null)
            .show()
    }

    private fun showConnectionOptions() {
        if (isDialogShowing) return
        val options = arrayOf("WiFi (Auto-detect)", "USB Tethering (Auto-detect)", "Manual IP Address")
        MaterialAlertDialogBuilder(this, R.style.GeneonDialogTheme)
            .setTitle("Connect Dashboard")
            .setAdapter(ArrayAdapter(this, R.layout.dialog_list_item, options)) { _, which ->
                isDialogShowing = false
                when (which) {
                    0 -> { Toast.makeText(this, "Searching via WiFi…", Toast.LENGTH_SHORT).show(); discoverServerViaUDP() }
                    1 -> { Toast.makeText(this, "Searching via USB…", Toast.LENGTH_SHORT).show(); discoverServerViaUsbTethering() }
                    2 -> showManualIpInputDialog()
                }
            }
            .setOnDismissListener { isDialogShowing = false }
            .create()
            .also { it.show(); isDialogShowing = true }
    }

    override fun onDestroy() {
        sensorManager.unregisterListener(this)
        super.onDestroy()
        statusUpdateJob?.cancel()
        closeConnection()
    }

    private fun setupSensors() {
        sensorManager = getSystemService(Context.SENSOR_SERVICE) as SensorManager
        accelerometer = sensorManager.getDefaultSensor(Sensor.TYPE_ACCELEROMETER)
    }

    override fun onResume() {
        super.onResume()
        accelerometer?.let {
            sensorManager.registerListener(this, it, SensorManager.SENSOR_DELAY_GAME)
        }
    }

    override fun onPause() {
        super.onPause()
        sensorManager.unregisterListener(this)
    }

    override fun onSensorChanged(event: SensorEvent?) {
        if (event == null) return
        if (event.sensor.type == Sensor.TYPE_ACCELEROMETER) {
            val x = event.values[0]
            val y = event.values[1]
            val z = event.values[2]

            val rawAngle = Math.toDegrees(Math.atan2(y.toDouble(), x.toDouble())).toFloat()

            if (isFirstReading) {
                lastRawAngle = rawAngle
                continuousAngle = rawAngle
                isFirstReading = false
            } else {
                var delta = rawAngle - lastRawAngle
                if (delta > 180) delta -= 360
                else if (delta < -180) delta += 360
                continuousAngle += delta
                lastRawAngle = rawAngle
            }

            // Invert the angle to match screen orientation
            val targetAngle = -continuousAngle
            
            // Apply smoothing
            smoothedHudAngle += (targetAngle - smoothedHudAngle) * HUD_ALPHA

            // Apply rotation to HUD elements
            centerGearBlock.rotation = smoothedHudAngle
        }
    }

    override fun onAccuracyChanged(sensor: Sensor?, accuracy: Int) {}
}
